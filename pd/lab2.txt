import java.util.Scanner;

public class Main {
    public static int silnia(int n) {
        int iloczyn = 1;
        for(int i = 1; i <= n; i++) {
            iloczyn *= i;
        }
        return iloczyn;
    }
    public static void main(String[] args) { 
    Scanner x = new Scanner(System.in);
    int n = x.nextInt();
    double[] liczby = new double[n];
    for(int i=0;i<n;i++)
    liczby[i]= x.nextInt();
    int A = 0;
    int B = 0;
    int C = 0;
    int D = 0;
    int E = 0;
    int F = 0;
    int G = 0;
    int H = 0;
    for(int i=0;i<n;i++)
    {
        if(liczby[i]%2!=0)
            A++;
        if(liczby[i]%3==0 && liczby[i]%5!=0)
            B++;
        if(liczby[i]*liczby[i]%2==0)
            C++;
        if(liczby[i]%2!=0 && liczby[i]>0)
            G++;
    }
    
    for(int i=1;i<n-1;i++)
    {
        if(liczby[i]<(liczby[i-1]+liczby[i+1]/2))
        D++;

        if(liczby[i]>Math.pow(2,i) && liczby[i]<silnia(i))
        E++;

        if(i%2!=0 && liczby[i]%2==0)
        F++;

        if(Math.abs(liczby[i])<Math.pow(i,2));
        H++;
    }
    System.out.println("A)"+A);
    System.out.println("B)"+B);
    System.out.println("C)"+C);
    System.out.println("D)"+D);
    System.out.println("E)"+E);
    System.out.println("F)"+F);
    System.out.println("G)"+G);
    System.out.println("H)"+H);
    x.close();
}
}
